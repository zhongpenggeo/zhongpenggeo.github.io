1. 右边纵坐标  
> gmt psscale -Dx5.2i/0.1i+w13c/0.3c -C$CPT -I -Bx50 -By+l"mGal" -O >>$PS
2. 下边横坐标
> gmt psscale -Dx0.2i/10.1i+w13c/0.3c+h -C$CPT -I -Bx50 -By+l"mGal" -O >>$PS

> gmt psscale -$RR -$JJ -DjTR+o-1c/1c+w4i/0.3c -C$CPT  -Bx50 -By+l"  mGal" -O -V >>$PS
